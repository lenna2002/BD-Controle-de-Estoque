CREATE TABLE produtos (
	codigo_produto INT PRIMARY KEY,
	nome_produto VARCHAR(40) NOT NULL,
	descricao TEXT NULL,
	preco NUMERIC CHECK(preco > 0) NOT NULL,
	quantidade_estoque SMALLINT DEFAULT 0,
	perecivel BOOLEAN NOT NULL
)
